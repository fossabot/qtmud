[docs]def roll(roller, dice_pool):
    result = 0
    for dice in range(dice_pool):
        result += random.choice([-1, 0, 0, 0, 1, 1])
    qtmud.log.debug('{} rolled a {} using {} dice'.format(roller.name,
                                                          result,
                                                          dice_pool))
    return result
