[docs]    def apply(self, thing):
        """
            .. versionadded:: 0.0.3-feature/learning
        """
        if not hasattr(thing, 'noises'):
            thing.noises = self.noises
        thing.manager.services[Noisemaker].add(thing)
        return thing
